generic aliases are beta-reduced when applied in an assertion

=======

type Id(T) as T
type Pair(A, B) as [A, B]

unittest "generics" do
  assert Id(1) <: number
  assert Id(string) == string
  assert Pair(1, string) <: [number, string]
  assert not (Pair(1, string) <: [string, string])
end

=======

type Id<T> = T;

type Pair<A, B> = [A, B];

=======
unittest "generics"
  ok      Id(1) <: number
  ok      Id(string) == string
  ok      Pair(1, string) <: [number, string]
  ok      not (Pair(1, string) <: [string, string])

4 assertion(s): 4 passed, 0 failed
