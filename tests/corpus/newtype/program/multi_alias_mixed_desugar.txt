Several aliases mixing let, cond, and not each desugar independently

=======

type A do let x = 1, y = 2 in [x, y] end

type B do cond do
    a <: b -> c,
    d <: e -> f,
    else -> g
end
end

type C do if not (a <: b) then c else d end end

=======

type A do [1, 2] end

type B do
    if a <: b then c
    else
        if d <: e then f
        else g
        end
    end
end

type C do if a <: b then d else c end end
