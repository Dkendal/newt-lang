An `and` condition in a type alias nests into two ifs

=======

type A do
    if a <: b and c <: d then
        e
    else
        f
    end
end

=======

type A do
    if a <: b then
        if c <: d then
            e
        else
            f
        end
    else
        f
    end
end

=======
