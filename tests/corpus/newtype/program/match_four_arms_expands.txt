A four-arm match inside a type alias nests four ifs

=======

type A do match T do
    number -> 1,
    string -> 2,
    boolean -> 3,
    else -> 4
end
end

=======

type A do
    if T <: number then 1
    else
        if T <: string then 2
        else
            if T <: boolean then 3
            else 4
            end
        end
    end
end

=======
